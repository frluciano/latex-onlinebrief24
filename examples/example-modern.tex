\documentclass[modern, footercenter]{onlinebrief24}

\setfromfirstname{John}
\setfromlastname{Doe}
\setfromaddress{Example Street 1 | 12345 Example City}
\setfromlandline{0123 / 456 788}
\setfromphone{0123 / 456 789}
\setfromemail{john.doe@example.com}
\setfromweb{www.example.com}
\setfromlinkedin{john-doe}

\setreturnaddress{John Doe, Example Street 1, 12345 Example City}
\setrecipient{Jane Doe \\ Example Avenue 456 \\ 67890 Example City}

\setsubject{Test of the new modern style}
\setplace{Example City}
\setdate{\today}

\begin{document}
\begin{letter}{}
    \opening{Dear Mrs. Doe,}
    
    this is a sample letter demonstrating the new \texttt{modern} option of the \texttt{onlinebrief24.cls} class.
    
    As you can see, my name appears in the top right in a large, modern font and my contact details can be found in the bottom right.
    
    Most importantly, the position of the address field and the return address for the window envelope remain exactly the same as in the standard version.
    
    \closing{Sincerely}
\end{letter}
\end{document}
